let listaAtual = [];
let indexAtual = 0;
let timerControles;
let timerSidebar; // Novo temporizador para a lateral

function iniciarFluxo(categoria) {
    document.getElementById('start-screen').style.display = 'none';
    listaAtual = listas[categoria];
    document.getElementById('sidebar-titulo').innerText = categoria.toUpperCase();
    indexAtual = 0;
    montarListaLateral();
    carregarCanais();
}

function carregarCanais() {
    const player = document.getElementById('main-player');
    if (listaAtual.length > 0) {
        player.src = listaAtual[indexAtual];
        marcarAtivoNaLista();
    }
}

function mudarCanal(direcao) {
    if (listaAtual.length === 0) return;
    indexAtual = (indexAtual + direcao + listaAtual.length) % listaAtual.length;
    carregarCanais();
}

function montarListaLateral() {
    const container = document.getElementById('lista-canais-container');
    container.innerHTML = "";
    listaAtual.forEach((url, i) => {
        let nomeRaw = url.split('/').pop().replace('?id=', '').replace('.php', '');
        if(nomeRaw.includes('canal=')) nomeRaw = nomeRaw.split('=')[1];
        const div = document.createElement('div');
        div.className = 'canal-item';
        div.id = 'item-' + i;
        div.innerText = `${i + 1}. ${nomeRaw.toUpperCase()}`;
        div.onclick = (e) => {
            e.stopPropagation();
            indexAtual = i;
            carregarCanais();
            resetarTimerSidebar(); // Reseta ao clicar em um canal
        };
        container.appendChild(div);
    });
}

function marcarAtivoNaLista() {
    document.querySelectorAll('.canal-item').forEach(el => el.classList.remove('ativo'));
    const ativo = document.getElementById('item-' + indexAtual);
    if(ativo) {
        ativo.classList.add('ativo');
        ativo.scrollIntoView({ behavior: 'smooth', block: 'center' });
    }
}

// Lógica de fechamento automático da Sidebar
function toggleSidebar() {
    const sb = document.getElementById('sidebar');
    sb.classList.toggle('aberto');
    
    if(sb.classList.contains('aberto')) {
        resetarTimerSidebar();
    } else {
        clearTimeout(timerSidebar);
    }
}

function resetarTimerSidebar() {
    clearTimeout(timerSidebar);
    const sb = document.getElementById('sidebar');
    
    if(sb.classList.contains('aberto')) {
        timerSidebar = setTimeout(() => {
            sb.classList.remove('aberto');
        }, 30000); // 30 segundos
    }
}

function voltarHome() {
    document.getElementById('sidebar').classList.remove('aberto');
    clearTimeout(timerSidebar);
    document.getElementById('main-player').src = "";
    document.getElementById('start-screen').style.display = 'flex';
}

function toggleMenu() {
    const ctrls = document.getElementById('ctrls');
    const catcher = document.getElementById('click-catcher');
    ctrls.classList.add('ativo');
    catcher.style.display = 'none';
    clearTimeout(timerControles);
    timerControles = setTimeout(() => {
        ctrls.classList.remove('ativo');
        catcher.style.display = 'block';
    }, 10000);
}
